// Exemplo lógico do componente de Sincronização
const OrderStatusRow = ({ orderId }) => {
  const [progress, setProgress] = useState(0);

  useEffect(() => {
    // Escuta o evento vindo do microfrontend de estoque
    socket.on(`receipt_update_${orderId}`, (data) => {
      setProgress(data.percentageCompleted);
    });
  }, [orderId]);

  return (
    <div className="row">
      <span>Pedido #{orderId}</span>
      <div className="progress-bar" style={{ width: `${progress}%` }} />
      <span>{progress}% Bipado</span>
    </div>
  );
};
